#include<iostream>
using namespace std;

const int N=10;

int n;
int path[N];
bool st[N];

void dfs (int u){ //u 表示层数
    if (u==n){
        for (int i=0;i<n;i++)
        cout<<path[i]<<" ";
        
        cout<<endl;
    }
    for (int i=1;i<=n;i++){//循环：从左到右遍历  //注意：i开始位置
        if(!st[i]){
            path[u]=i;
            st[i]=true;
            dfs(u+1); //递归：从上到下遍历
            // path[u]=0;//无用、一定会被覆盖 可删除
            
            st[i]=false;//注意恢复修改的内容
            
        }
        
    }
}

int main(){
    cin>>n;
    
    dfs(0);
}

#include<iostream>
using namespace std;

const int N=20;//对角线个数2n-1

int n;
char g[N][N];
bool col[N],dg[N],udg[N]; //以行为索引，仅需要开辟: 列数组及正、反对角线数组

void dfs (int u){ //u 表示层数
    if (u==n){
        for (int i=0;i<n;i++)
        cout<<g[i]<<endl;
        cout<<endl;
        
    }
    for (int i=0;i<n;i++){//循环：从左到右遍历(列遍历)  //注意：i开始位置
        if(!col[i] && !dg[u+i] && !udg[i-u+n]){
            g[u][i]='Q';
            col[i] = dg[u+i] = udg[i-u+n] = true;
            dfs(u+1); //递归：从上到下遍历（行遍历）
            // path[u]=0;//无用、一定会被覆盖 可删除
            col[i] = dg[u+i] = udg[i-u+n] = false;//注意恢复修改的内容
            g[u][i]='.';
        }
        
    }
}

int main(){
    cin>>n;
    
    for (int i=0;i<n;i++)
     for (int j=0;j<n;j++)
        g[i][j]='.';
    
    dfs(0);
}

#include<iostream>
using namespace std;

const int N=20;

int n;
char g[N][N];
bool row[N],col[N],dg[N],udg[N]; 

void dfs (int x, int y, int s){
    
    if (y==n) y=0,x++;
    if (x==n){
        if (s==n){
            for (int i=0;i<n;i++)
            cout<<g[i]<<endl;
            cout<<endl;
        }
        return; //无所谓？
    }
    
    dfs(x,y+1,s);
    
    if(!row[x] && !col[y] && !dg[x+y] && !udg[x-y+n]){
        g[x][y]='Q';
        row[x] = col[y] = dg[x+y] = udg[x-y+n] = true;
        dfs(x,y+1,s+1); //递归：从上到下遍历（行遍历）
        // path[u]=0;//无用、一定会被覆盖 可删除
        row[x] = col[y] = dg[x+y] = udg[x-y+n] = false;//注意恢复修改的内容
        g[x][y]='.';
    }


}

int main(){
    cin>>n;
    
    for (int i=0;i<n;i++)
     for (int j=0;j<n;j++)
        g[i][j]='.';
    
    dfs(0,0,0);
}

#include<iostream>
#include<cstring>

using namespace std;

typedef pair<int, int> PII;

const int N=110;

int n,m;

int g[N][N];//迷宫
int d[N][N];//距离
PII q[N*N];//带扩展点位置(模拟队列)

int bfs()
{
    int hh=0,tt=0;
    q[0]={0,0};
    
    memset(d,-1,sizeof d);
    d[0][0]=0;
    
    int dx[4]={-1, 0, 1, 0}, dy[4]={0, 1, 0, -1};
    
    while(hh<=tt)
    {
        auto t=q[hh++];
        
        for(int i=0; i<4; i++)
        {
            int x=t.first + dx[i];
            int y=t.second + dy[i];
            
            if(x >= 0 && x < n && y >=0 && y <m && g[x][y]==0 && d[x][y]==-1) //扩展条件判断
            {
                d[x][y]=d[t.first][t.second]+1;
                q[++tt]={x,y}; //待扩展目标添加
            }
        }
        
    }
    
    return d[n - 1][m - 1];
    
}

int main(){
    cin>>n>>m;
    for(int i=0;i<n;i++)
        for (int j=0;j<m;j++)
         cin>>g[i][j];
    
    cout<<bfs()<<endl;
    
    return 0;
}
