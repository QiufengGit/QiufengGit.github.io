void quick_sort(int q[],int l, int r){
    if(l>=r) return;
    int i=l-1,j=r+1; //初始位置
    
    int x=q[l+r>>1]; //中间值位置 选择j时不变，选择i时l+r+1 （边界问题）
    
    while(i<j){ //逐次处理
        do i++; while(q[i]<x); //注意起点位置
        do j--;while(q[j]>x);
        if(i<j) swap(q[i],q[j]); //单次交换
    }
    
    quick_sort(q,l,j); //递归处理子区间
    quick_sort(q,j+1,r);
}

注：

  1. while(i<j)不能写成 i<=j,这样会多做一次循环，导致跳出循环时i>j。跳出循环时i应该和j相等。
  2. 注意i、j模板与初始分界点x的关系。
  3. 如果初始分界点x取左端点或者右端点，在所有数都一样的情况下，时间复杂度会退化到 O($n^2$)，原为$n*log_2(n)$

1.2 归并

思路：分治

  1. 找出分界点，分界点取数组的中间位置（与快速排序不同，分界点取中间位置的数值）。
  2. 分界后对两边做递归排序。
  3. 归并，合二为一。（从小到大则是取较小值放到tmp中，从大到小则是取较大值放到tmp中）
  4. 时间复杂度：$n*log_2(n)$。（n对2取对数）

void merge_sort(int q[],int l,int r){
   if(l>=r) return;
   int mid=l+r>>1; //边界点确定
    merge_sort(q,l,mid); //递归
    merge_sort(q,mid+1,r);
    int k=0,i=l,j=mid+1; //两边初值
    while(i<=mid&&j<=r){
        if(q[i]<q[j]) //取小值赋值
            tmp[k++]=q[i++];
        else
            tmp[k++]=q[j++];
    }
    while(i<=mid) //取剩余
        tmp[k++]=q[i++];
    while(j<=r)
        tmp[k++]=q[j++];
    for (i=l,j=0;i<=r;i++,j++) q[i]=tmp[j]; //排序后赋值 从l到r
}

注：

  1. while(i<=mid&&j<=r) 两边循环到底
  2. 注意i、j模板与初始分界点mid的关系。
  3. 时间复杂度稳定为$n*log_2(n)$
  4. 与快排不同，需要空间存储，故数据量小时适合归并

// 区间[l, r]被划分成[l, mid]和[mid + 1, r]时使用：
int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;    // check()判断mid是否满足性质
        else l = mid + 1;
    }
    return l;
}
// 区间[l, r]被划分成[l, mid - 1]和[mid, r]时使用：
int bsearch_2(int l, int r)
{
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

double bsearch_3(double l, double r)
{
    const double eps = 1e-6;   // eps 表示精度，取决于题目对精度的要求
    while (r - l > eps)
    {
        double mid = (l + r) / 2;
        if (check(mid)) r = mid;
        else l = mid;
    }
    return l;
}

// C = A + B, A >= 0, B >= 0
vector<int> add(vector<int> &A, vector<int> &B)
{
    if (A.size() < B.size()) return add(B, A);

    vector<int> C;
    int t = 0;
    for (int i = 0; i < A.size(); i ++ )
    {
        t += A[i];
        if (i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }

    if (t) C.push_back(t);
    return C;
}

// C = A - B, 满足A >= B, A >= 0, B >= 0
vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    for (int i = 0, t = 0; i < A.size(); i ++ )
    {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

vector<int> mul(vector<int> &A, int b)
{
    vector<int> C;

    int t = 0;
    for (int i = 0; i < A.size() || t; i ++ )
    {
        if (i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }

    while (C.size() > 1 && C.back() == 0) C.pop_back();

    return C;
}

// A / b = C ... r, A >= 0, b > 0
vector<int> div(vector<int> &A, int b, int &r)
{
    vector<int> C;
    r = 0;
    for (int i = A.size() - 1; i >= 0; i -- )
    {
        r = r * 10 + A[i];
        C.push_back(r / b);
        r %= b;
    }
    reverse(C.begin(), C.end());
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    return C;
}

4.2 差分

思路：

  1. 前缀和的逆运算

  2. a[i]=b[1]+b[2]+b[3]+···+b[i]，则称b为a的差分数组，a为b的前缀和，且b满足：

​ b[1]=a[1];

​ b[2]=a[2]-a[1];

​ b[3]=a[3]-a[2];

​ b[n]=a[n]-a[n-1];

  1. a[i]+c,对b数组的影响是b[i]+c，并且b[i+1]-c。(从a[i]=b[1]+b[2]+b[3]+···+b[i]中可以得到）

​ a数组的第l到r个数＋c，对b数组的影响是b[l]+c并且b[r+1]-c。

步骤：

  1. 确定插入方式
  2. 构造原始序列（0序列满足差分及前缀和）
  3. 插入获取目标序列（同时生成相应差分序列）

#include <iostream>
using namespace std;

const int N=1010;
int a[N][N],b[N][N];

void insert(int x1,int y1,int x2,int y2,int c){
    //1.前缀和a矩阵在(x1,y1)到(x2,y2)的范围整体加c，对b的影响
    b[x1][y1]+=c;
    b[x1][y2+1]-=c;
    b[x2+1][y1]-=c;
    b[x2+1][y2+1]+=c;
}
int main(){
    int n,m,q;
    scanf("%d%d%d",&n,&m,&q);
    //1.读入数据
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            scanf("%d",&a[i][j]);
        }
    }
    //2.初始化差分数组
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            insert(i,j,i,j,a[i][j]);
        }
    }
    //3.循环处理
    while(q--){
        int x1,y1,x2,y2,c;
        cin>>x1>>y1>>x2>>y2>>c;
        insert(x1,y1,x2,y2,c);
    }
    //4.得到b的前缀和a
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            a[i][j]=b[i][j]+a[i-1][j]+a[i][j-1]-a[i-1][j-1];
            cout<<a[i][j]<<' ';
        }
        cout<<endl;
    }
    return 0;
}

typedef pair<int,int> PII;

vector<PII> add,query;//add存储插入的数对x,c query存储查询的数对l,r
vector<int> alls;//存放所有出现的下标，包括插入的x和查询的左右边界l r

int find(int x){

    while(l<r){
        int mid = l + r >> 1;
        if(alls[mid]>=x) r=mid;
        else l=mid+1;

}

        alls.push_back(r);//记得把左右边界的下标也一并记录
    }
    //3.对下标进行排序并去重
    sort(alls.begin(),alls.end());
    alls.erase(unique(alls.begin(),alls.end()),alls.end());
    //4.离散化
    //离散化就是把大而分散的一段段使用到的稀疏区间，整合映射到连续的一段较小的稠密区间里
    for(auto item:add){
        int x=find(item.first);
        a[x]+=item.second;
    }
    //5.预处理前缀和
    for(int i=1;i<=alls.size();i++) s[i]=s[i-1]+a[i];
    //6.开始查询
    for(auto item:query){
        int l=find(item.first),r=find(item.second);//先找边界映射的下标
        int sum = s[r] - s[l-1];
        cout<<sum<<endl;
    }
}

// 将所有存在交集的区间合并
void merge(vector<PII> &segs)
{
    vector<PII> res;

    sort(segs.begin(), segs.end());//排序

    int st = -2e9, ed = -2e9; //起始区间（前一个区间）
    for (auto seg : segs)
        if (ed < seg.first) //区间没有交集则 独立并剔出 前一个区间
        {
            if (st != -2e9) res.push_back({st, ed});
            st = seg.first, ed = seg.second;
        }
        else ed = max(ed, seg.second); //存在交集则 扩大前一个区间 但不剔出

    if (st != -2e9) res.push_back({st, ed}); //加入最后一个区间

    segs = res;
}

#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

typedef pair<int,int> PII;
vector<PII> segs;

void merge(vector<PII> &segs){
    vector<PII> res;//用于存储合并后的区间
    //1.将区间按照左端点大小升序排序（默认按照segs.first排序）
    sort(segs.begin(),segs.end());
    //2.开始合并
    int st = -2e9,ed=-2e9;//st ed存储上一个区间的左右端点
    for(auto seg :segs){
        //如果当前区间的左端点大于上一个区间的右端点 说明上一个区间独立了
        if(seg.first > ed){ 
            if(st!=-2e9) res.push_back({st,ed});//第一次循环不执行
            //把当前区间作为要维护的下一个区间
            st = seg.first, ed = seg.second; 
        }
        //如果当前区间的左端点不大于上一个区间的右端点 那么合并当前和上一个区间
        else ed = max(ed,seg.second);
    }
    if(st!=-2e9) res.push_back({st,ed});//记得把最后一个区间的结果加进去
    segs = res;
}

        cin>>l>>r;
        segs.push_back({l,r});
    }
    merge(segs);
    cout<<segs.size()<<endl;
}
